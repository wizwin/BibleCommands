-------------------------------------------------------------------------------
Report, Save and Restore Windows drivers (Needs Admin)
-------------------------------------------------------------------------------

Report
------
Driverquery /v /fo table > DriverTable.doc

Save
----
dism /online /export-driver /destination:"Full path"

Restore
-------
pnputil /add-driver "Full path\*.inf" /subdirs /install /reboot

-------------------------------------------------------------------------------
Ping a range of IP addresses
-------------------------------------------------------------------------------

FOR /L %x IN (1,1,254) DO ping -n 1 192.168.1.%x

-------------------------------------------------------------------------------
7-Zip all files in a directory
-------------------------------------------------------------------------------

for /d %i in (*.*) do 7z a -r "%i.zip" "%i"

-------------------------------------------------------------------------------
Cleanup Windows Installer
-------------------------------------------------------------------------------

Dism.exe /online /Cleanup-Image /StartComponentCleanup

-------------------------------------------------------------------------------
Save Metadata from Videos to separate files (xml, txt)
-------------------------------------------------------------------------------

mediainfo --Output=PBCore2 -f MVI_1000.MP4 > MVI_1000.xml
mediainfo -f MVI_1000.MP4 > MVI_1000.txt
